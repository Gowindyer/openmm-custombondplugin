set(OPENMMCUSTOMBONDREFERENCE_LIBRARY_NAME OpenMMCustomBondReference)

set(SOURCE_FILES
    ${CMAKE_CURRENT_SOURCE_DIR}/src/CustomBondReferenceKernelFactory.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/src/CustomBondReferenceKernels.cpp
)
set(SOURCE_INCLUDE_FILES
    ${CMAKE_CURRENT_SOURCE_DIR}/include/CustomBondReferenceKernelFactory.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/CustomBondReferenceKernels.h
)

add_library(${OPENMMCUSTOMBONDREFERENCE_LIBRARY_NAME} SHARED ${SOURCE_FILES} ${SOURCE_INCLUDE_FILES})
target_include_directories(${OPENMMCUSTOMBONDREFERENCE_LIBRARY_NAME}
    PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/include
        ${CMAKE_CURRENT_SOURCE_DIR}/src
        ${CMAKE_SOURCE_DIR}/openmmapi/include
        ${OPENMM_INCLUDE_DIR}
        ${OPENMM_REFERENCE_INCLUDE_DIR}
        ${OPENMM_ASMJIT_INCLUDE_DIR}
)
target_compile_definitions(${OPENMMCUSTOMBONDREFERENCE_LIBRARY_NAME} PRIVATE OPENMM_BUILDING_SHARED_LIBRARY)
target_link_libraries(${OPENMMCUSTOMBONDREFERENCE_LIBRARY_NAME}
    PRIVATE
        ${OPENMM_CUSTOMBOND_TARGET}
        ${OPENMM_LIBRARY}
)
set_target_properties(${OPENMMCUSTOMBONDREFERENCE_LIBRARY_NAME} PROPERTIES
    BUILD_RPATH "${OPENMM_CUSTOMBOND_PLUGIN_RPATH}"
    INSTALL_RPATH "${OPENMM_CUSTOMBOND_PLUGIN_RPATH}"
)
install(TARGETS ${OPENMMCUSTOMBONDREFERENCE_LIBRARY_NAME}
    LIBRARY DESTINATION lib/plugins
    RUNTIME DESTINATION lib/plugins
    ARCHIVE DESTINATION lib/plugins
)

if(OPENMM_CUSTOMBOND_BUILD_TESTS AND BUILD_TESTING)
    add_subdirectory(tests)
endif()
